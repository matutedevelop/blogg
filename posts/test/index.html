use std::collections::HashSet;

fn is_valid(s: String) -> bool {
    let mut map = HashSet::new();
    s.chars().filter(|c| !")]}".contains(*c)).for_each(|c| {
        map.insert(c);
    });
    s.chars()
        .filter(|c| ")]}".contains(*c))
        .for_each(|c| match c {
            _ if c == ')' => {
                map.remove(&'(');
            }
            _ if c == ']' => {
                map.remove(&'[');
            }
            _ if c == '}' => {
                map.remove(&'{');
            }
            _ => (),
        });
    map.len() == 0
}
